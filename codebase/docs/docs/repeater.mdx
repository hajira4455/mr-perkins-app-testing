### Animated

---

<p>
  We're using{' '}
  <a href='https://github.com/frankwallis/react-slidedown' target='_blank'>
    react-slidedown
  </a>{' '}
  to animate the cloning.
</p>

```jsx
import Repeater from '@components/repeater'

const RepeatingForm = () => {
  const [count, setCount] = useState(1)

  const increaseCount = () => {
    setCount(count + 1)
  }

  const deleteForm = e => {
    e.preventDefault()
    const slideDownWrapper = e.target.closest('.react-slidedown'),
      form = e.target.closest('form')
    if (slideDownWrapper) {
      slideDownWrapper.remove()
    } else {
      form.remove()
    }
  }

  return (
    <Repeater count={count}>
      {i => {
        const Tag = i === 0 ? 'div' : SlideDown
        return (
          <Tag key={i}>
            <Form>...</Form>
          </Tag>
        )
      }}
    </Repeater>
  )
}
```
