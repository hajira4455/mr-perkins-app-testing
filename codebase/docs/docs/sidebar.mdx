### Default

---

<p>
  Creating a sidebar component is easy all you need is <code>open</code>, <code>toggleSidebar</code> &{' '}
  <code>title</code> props.
</p>

<p>
  You can check demo on{' '}
  <a href='https://pixinvent.com/demo/vuexy-react-admin-dashboard-template/demo-1/apps/invoice/preview' target='_blank'>
    this
  </a>{' '}
  page.
</p>

```jsx
import Sidebar from '@components/sidebar'

const component = () => (
  <Sidebar
    size='lg'
    open={open}
    title='Add Payment'
    headerClassName='mb-1'
    contentClassName='p-0'
    toggleSidebar={toggleSidebar}
  >
    ...
  </Sidebar>
)
```
