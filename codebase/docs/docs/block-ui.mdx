### Basic

---

<p>
  Creating a ui loader is easy, all you have to do is wrap your component with it and pass <code>blocking</code> prop
</p>

```jsx
import UILoader from '@components/ui-loader'

const component = () => {
  const [block, setBlock] = useState(false)

  const handleBlock = () => {
    setBlock(true)

    setTimeout(() => {
      setBlock(false)
    }, 3000)
  }

  return <UILoader blocking={block}>...</UILoader>
}
```

### Custom Loader

---

<p>
  Use <code>loader</code> prop to change loading spinner.
</p>

```jsx
import UILoader from '@components/ui-loader'
import Spinner from '@components/spinner/Loading-spinner'

const component = () => {
  const [block, setBlock] = useState(false)

  const handleBlock = () => {
    setBlock(true)

    setTimeout(() => {
      setBlock(false)
    }, 3000)
  }

  return (
    <UILoader blocking={block} loader={<Spinner />}>
      ...
    </UILoader>
  )
}
```
